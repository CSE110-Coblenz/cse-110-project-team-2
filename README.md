    npm init -y

    # Install dependencies
    npm install konva react-konva react react-dom

    # Install TypeScript and typings
    npm install --save-dev typescript @types/react @types/react-dom vite

    # Ensure Vite and React plugin are installed
    npm install --save-dev vite @vitejs/plugin-react

    # Run server
    npx vite
